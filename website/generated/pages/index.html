Get Started in Seconds

1. Install 

  
  go get github.com/nxtcoder17/fastlog

2. Basic Usage 

  
  package main
          import (
          "github.com/nxtcoder17/fastlog"
          "log/slog"
          )
          func main() {
          // Create a console logger
          logger := fastlog.New(
          fastlog.Console(),
          fastlog.WithLogLevel(slog.LevelInfo),
          )
          // Log with native API
          logger.Info("application started")
          logger.Debug("debugging", "enabled", true)
          // Or use as slog handler
          slogger := slog.New(logger.SlogHandler())
          slogger.Info("using slog API", "version", "1.0")
          }

3. Advanced Configuration 

  
  // JSON output
        jsonLogger := fastlog.New(

        // With context
        userLogger := logger.With("user_id", 12345)
        userLogger.Info("user action", "action", "login")
